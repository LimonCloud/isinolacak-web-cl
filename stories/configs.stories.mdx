# Configs

##### Example Themes Config

```tsx
import {
    IOCoreProvider,
    Container,
    useIOCoreTheme
} from "isinolacak-web";

// If you want it to auto suggest the values you added, you can do it like this:
declare namespace IOCore {
	interface Colors {
        myColor: string;
    }

    interface Typography {
        myFont: IOCore.TypographyType
    }
};

const MOCK_THEMES = [
    {
        key: "light",
        typography: {
            body: {
                fontFamily: "Montserrat-Black",
                lineHeight: 1.1,
                fontSize: 16
            },
            header1: {
                fontFamily: "Montserrat-Regular",
                lineHeight: 2,
                fontSize: 50,
	        },
            myFont: {
                fontFamily: "My-Font",
                lineHeight: 1.2,
                fontSize: 24
            }
        },
        colors: {
            primary: "black",
            secondary: "white",
            myColor: "#4d78ca"
        }
    }
];

const NavigationHomePage = () => {
    const {
        typography,
        colors
    } = useIOCoreTheme();

    return <Container>
        <span
            style={{
                color: colors.myColor,
                ...typography["myFont"]
            }}
        >
            Hello
        </span>
    </Container>;
};

const App = () => {
    return <IOCoreProvider
        configs={{
            themes: MOCK_THEMES,
            // You can send a another config at the same time.
        }}
    >
        <NavigationHomePage/>
    </IOCoreProvider>;
};
export default App;
```

```tsx
    </IOCoreProvider>;
};
export default App;
```

##### Example Locales Config

```tsx
import {
    useIOCoreLocale,
    IOCoreProvider,
    Container
} from "isinolacak-web";

// If you want it to auto suggest the values you added, you can do it like this:
declare namespace IOCore {
	interface Translation {
        "my-word-key": string;
    }
};

const MOCK_LOCALES = [
    "code": "es",
    "isRTL": "false",
    "translations": {
        "app-name": "IOCore Web",
        "my-word-key": "My Word"
    }
];

const NavigationHomePage = () => {
    const {
        localize
    } = useIOCoreLocale();

    return <Container>
        <span>
            {localize("my-word-key")}
        </span>
    </Container>;
};

const App = () => {
    return <IOCoreProvider
        configs={{
            locales: MOCK_LOCALES,
            // You can send a another config at the same time.
        }}
    >
        <NavigationHomePage/>
    </IOCoreProvider>;
};
export default App;
```
